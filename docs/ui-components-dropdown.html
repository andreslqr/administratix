Controlling from Livewire

You can control any dropdown of your form from livewire if you add the \Administratix\Administratix\Support\Livewire\Components\WithDropdowns trait in your component, then you can access to any method

Preview Code Livewire
  • Item 1
  • Item 2
        <x-admin::dropdown wire:dropdown="firstMenu">
            <x-slot name="trigger">
                <x-admin::button>
                    click
                </x-admin::button>
            </x-slot>
            <x-slot name="content" class="menu p-2 shadow bg-base-100 rounded-box w-52">
                <li><a>Item 1</a></li>
                <li><a>Item 2</a></li>
            </x-slot>
        </x-admin::dropdown>
    
        <?php

        class MyLivewireComponent extends Component
        {
            use \Administratix\Administratix\Support\Livewire\Components\WithDropdowns;
            
            /**
            * The initial data of the dropdown elements
            * 
            * @var array
            */
            public $dropdowns = [
                'firstMenu' => false
            ];

            public function editData()
            {
                $this->toggleDropdown('firstMenu'); // Toggle the dropdown open / close

                $this->showDropdown('firstMenu'); // Show the dropdown

                $this->hideDropdown('firstMenu'); // Hide the dropdown

                $this->getDropdown('firstMenu'); // Return true or false depending if the dropdown is open or not

                if($this->isDropdownOpen('firstMenu')) // Check if the menu is open
                    $this->doSomething();

                if($this->isDropdownClosed('firstMenu')) // Check if the menu is closed
                    $this->doSomenthing();
            }
        }
