Controlling from Livewire

You can control any collapse of your form from livewire if you add the \Administratix\Administratix\Support\Livewire\Components\WithCollapses trait in your component, then you can access to any method

Preview Code Livewire
Focus me to see content

i'm hidding

        <x-admin::collapse wire:collapse="menu">
            <x-slot name="trigger">
                Focus me to see content 
            </x-slot>
            <x-slot name="content">
                I'm hidding
            </x-slot>
        </x-admin::collapse>
    
        <?php

        class MyLivewireComponent extends Component
        {
            use \Administratix\Administratix\Support\Livewire\Components\WithCollapses;
            
            /**
            * The initial state of the collapse elements
            * 
            * @var array
            */
            public $collapses = [
                'menu' => false
            ];

            public function editData()
            {
                $this->toggleCollapse('menu'); // Toggle the collapse open / close

                $this->showCollapse('menu'); // Show the collapse

                $this->hideCollapse('menu'); // Hide the collapse

                $this->getCollapse('menu'); // Return true or false depending if the collapse is open or not

                if($this->isCollapseOpen('menu')) // Check if the menu is open
                    $this->doSomething();

                if($this->isCollapseClosed('menu')) // Check if the menu is closed
                    $this->doSomenthing();
            }
        }
