Controlling from Livewire

You can control any steps of your form from livewire if you add the \Administratix\Administratix\Support\Livewire\Components\WithSteppers trait in your component, then you can access to any method, the $steppers property defines the steps from a stepper, and the property $step defines the current step

        <x-admin::steps class-for-active="step-neutral" wire:stepper="payment">
            <x-slot name="register">
                Register here
            </x-slot>
            <x-slot name="login">
                Login here
            </x-slot>
            <x-slot name="fillForm">
                Fill Form here
            </x-slot>
            <x-slot name="selectPaymentMethod">
                Select payment method method here
            </x-slot>
            <x-slot name="pay">
                Pay here
            </x-slot>
        </x-admin::steps>
    
        <?php

        class MyLivewireComponent extends Component
        {
            use \Administratix\Administratix\Support\Livewire\Components\WithSteppers;
            
            /**
            * The stepper elements
            * 
            * @var array
            */
            public $steppers = [
                'payment' => [
                      'register',
                      'login',
                      'fill form',
                      'select payment method',
                      'pay'
                ]
            ];

            /**
            * The stepper instance initial state
            * 
            * @var array
            */
            public $step = [
                'payment' => 'register'
            ];

            public function editData()
            {
                $this->getStep('payment'); // get the current active step

                $this->setStep('payment', 'fill form'); // set the step to active

                $this->nextStep('payment'); // Show the next step

                $this->prevStep('payment'); // Show the previous step

                if($this->isInStep('payment', 'login')) // Check if the alert is open
                    $this->doSomething();

                if($this->stepWasPassed('payment', 'login')) // Check if the alert is closed
                    $this->doSomenthing();
            }
        }
