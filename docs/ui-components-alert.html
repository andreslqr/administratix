Controlling from Livewire

You can control any alert of your form from livewire if you add the \Administratix\Administratix\Support\Livewire\Components\WithAlerts trait in your component, then you can access to any method

Preview Code Livewire
12 unread messages. Tap to see. 
        <x-admin::button wire:click="toggleAlert('infoAlert')">open alert</x-admin::button>
        <x-admin::alert class="alert-info" wire:alert="infoAlert">
            <x-slot name="icon">
                <svg xmlns="http://www.w3.org/2000/svg" class="stroke-current flex-shrink-0 h-6 w-6" fill="none" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M9 12l2 2 4-4m6 2a9 9 0 11-18 0 9 9 0 0118 0z" /></svg>
            </x-slot>
            <x-slot name="content">
                <span>12 unread messages. Tap to see.</span>
            </x-slot>
        </x-admin::alert>
    
        <?php

        class MyLivewireComponent extends Component
        {
            use \Administratix\Administratix\Support\Livewire\Components\WithModals;
            
            /**
            * The initial state of the alerts elements
            * 
            * @var array
            */
            public $alerts = [
                'infoAlert' => false
            ];

            public function editData()
            {
                $this->toggleAlert('infoAlert'); // Toggle the alert open / close

                $this->showAlert('infoAlert'); // Show the alert

                $this->hideAlert('infoAlert'); // Hide the alert

                $this->getAlert('infoAlert'); // Return true or false depending if the alert is open or not

                if($this->isAlertOpen('infoAlert')) // Check if the alert is open
                    $this->doSomething();

                if($this->isAlertClosed('infoAlert')) // Check if the alert is closed
                    $this->doSomenthing();
            }
        }
